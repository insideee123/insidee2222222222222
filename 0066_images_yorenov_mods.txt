        addEventOnChangeHudInfo(callBack) {
            try {
                if (this.data.callBacks.length === 0) {
                    if (!window.interface ||
                        !window.interface("Hud") ||
                        !window.interface("Hud").info
                    ) {
                        throw new Error("Hud interface or info is undefined");
                    }
                    const hudInfo = JSON.parse(
                        JSON.stringify(window.interface("Hud").info)
                    );
                    window.interface("Hud").info = new Proxy(hudInfo, {
                        set(target, property, value) {
                            try {
                                yorenovHud.hud.data.callBacks.forEach((cb) =>
                                    cb(property, value)
                                );
                            } catch (error) {}
                            return Reflect.set(target, property, value);
                        },
                    });
                }
                this.data.callBacks.push(callBack);
            } catch (error) {
                yorenovHud.addChatMessage(`(error) Error in addEventOnChangeHudInfo: ${error}`);
            }

				if (e === "mileage") this.setMileage(a);
				if (e === "fuel") this.setFuel(a);
				if (e == "left") {
					this.data.turnsEls[0].classList.toggle("YORENOV-speedometer-turn--active", +a);
				}
				if (e == "right") {
					this.data.turnsEls[1].classList.toggle("YORENOV-speedometer-turn--active", +a);
				}
				if (e == "rem") {
					this.data.paramsEls[2].classList.toggle("YORENOV-spedometer-param--active", +a);
				}
				if (e == "doors") {
					this.data.paramsEls[3].classList.toggle("YORENOV-spedometer-param--active", +a);
				}
				if (e == "temperature") {
					this.data.paramsEls[1].classList.toggle("YORENOV-spedometer-param--active", +a);
				}
				if (e == "lights") {
					this.data.paramsEls[0].classList.toggle("YORENOV-spedometer-param--active", +a);
				}
				if (e === "tachometer-show") {
					const display = +a >= 1 ? "" : "none";
					this.data.tachometer.wrapper.style.display = display;
				}
				if (e === "tachometer") {
					this.setGear(a.gear);
					this.setTachometer(a.rpm, a.maxRpm);
				}
			} catch (error) {
				yorenovHud.addChatMessage(`Speedometer error: ${error}`);
			}
		},

		show() {
			this.data.speedometerEl.style.right = "0vh";
			this.data.speedometerEl.style.opacity = "1";
			const greenZone = document.querySelector(".YORENOV-hud__green-zone");
			if (greenZone) greenZone.style.bottom = "25vh";
		},

		hide() {
			this.data.speedometerEl.style.right = "-70vh";
			this.data.speedometerEl.style.opacity = "0";
			const greenZone = document.querySelector(".YORENOV-hud__green-zone");
			if (greenZone) greenZone.style.bottom = "5vh";
		},

		setGear(e) {
			const gearValue = this.getBigSizeValue(e);
			this.data.tachometer.gearEl.innerHTML = gearValue;
		},

		setTachometer(e, a) {
			if (window.interface("Hud").speedometer.tachometer.show && e / a > 0.5) {
				this.data.tachometer.barEl.src = `${window.yorenovSampHud.hud.speedometer.arrow_glow}`;
			} else {
				this.data.tachometer.barEl.src = `${window.yorenovSampHud.hud.speedometer.arrow}`;
			}
			if (e > a) e = a;
			this.data.tachometer.barEl.style.transform = `rotate(${e / a * 327 + 180}deg)`;
		},

		setMileage(e) {
			const mileageValue = this.getBigSizeValue(e);
			this.data.mileageEl.innerHTML = mileageValue;
		},

		setSpeed(e) {
			const speedValue = this.getBigSizeValue(e);
			this.data.speed.valueEl.innerHTML = speedValue;
			if (e > 360) e = 360;
			const rotation = e / 360 * this.barsValues.speed.dashArray;
			this.data.speed.barEl.style.transform = `rotate(${rotation}deg) scale(-1)`;
		},

		setDamage(e) {
			this.data.damage.progress.style.width = `${Math.floor(e * 100)}%`;
		},

		setWash(e) {
			this.data.wash.progress.style.width = `${Math.floor(e * 100)}%`;
		},

		setFuel(e) {
			const fuelValue = this.getSmallSizeValue(e);
			const percentage = e / window.interface("Hud").speedometer.maxFuel * 100;
			let color = "#3ACB3B";

			if (percentage <= 5) color = "#DB2C22";
			else if (percentage <= 25) color = "#DB4222";
			else if (percentage <= 35) color = "#DB7122";
			else if (percentage <= 50) color = "#DB9C22";
			else if (percentage <= 65) color = "#CFC023";
			else if (percentage <= 80) color = "#9FC42B";
			else if (percentage <= 90) color = "#6CC733";

			this.data.fuel.valueEl.innerHTML = fuelValue;
			this.data.fuel.progress.style.width = `${percentage}%`;
			this.data.fuel.progress.style.backgroundColor = color;
		},

		getSmallSizeValue(e) {
			const digits = e.toString().split("");
			let result = "";
			for (let i = 0; i < digits.length; i += 1) {
				result += `<img class="YORENOV-digit YORENOV-digit-${digits[i]} YORENOV-digit-small" src="${window.yorenovSampHud.hud.speedometer.digits[digits[i]]}" alt="">`;
			}
			return result;
		},

		getBigSizeValue(e) {
			const digits = e.toString().split("");
			let result = "";
			for (let i = 0; i < digits.length; i += 1) {
				result += `<img class="YORENOV-digit YORENOV-digit-${digits[i]} YORENOV-digit-big" src="${window.yorenovSampHud.hud.speedometer.digits[digits[i]]}" alt="">`;
			}
			return result;
		},

		create(e) {
			this.data.speedometerEl = e;
			this.data.speed.valueEl = e.querySelector(".YORENOV-speedometer-speed__value");
			this.data.speed.barEl = e.querySelector(".YORENOV-speedometer-speed__arrow");
			this.data.fuel.valueEl = e.querySelector(".YORENOV-speedometer-fuel__value");
			this.data.fuel.progress = e.querySelector(".YORENOV-fuel-bar__value");
			this.data.wash.progress = e.querySelector(".YORENOV-wash-bar__value");
			this.data.wash.valueEl = e.querySelector(".YORENOV-speedometer-wash__value");
			this.data.damage.progress = e.querySelector(".YORENOV-damage-bar__value");
			this.data.damage.valueEl = e.querySelector(".YORENOV-speedometer-damage__value");
			this.data.mileageEl = e.querySelector(".YORENOV-speedometer-mileage__value");
			this.data.paramsEls = e.querySelector(".YORENOV-speedometer__params").children;
			this.data.turnsEls = e.querySelector(".YORENOV-speedometer__turns").children;
			this.data.tachometer.barEl = e.querySelector(".YORENOV-tachometer__arrow");
			this.data.tachometer.wrapper = e.querySelector(".YORENOV-speedometer__tachometer");
			this.data.tachometer.gearEl = this.data.tachometer.wrapper.querySelector("#YORENOV-tachometer-gear");
		},

		init() {
			const html = yorenovHud.getHtmlSpeedometer();
			const element = this._createSpeedometerElement(html, "YORENOV-speedometer");
			this.create(element);

			//   
			if (window.interface && window.interface("Hud")) {
				window.interface("Hud").speedometer.show = 0;
			}
		},

		_createSpeedometerElement(html, className) {
			const element = document.createElement("div");
			try {
				//    speedometer
				if (window.interface && window.interface("Hud")) {
					const speedometer = JSON.parse(JSON.stringify(window.interface("Hud").speedometer));
					window.interface("Hud").speedometer = new Proxy(speedometer, {
						set: (target, property, value) => {
							yorenovHud.speedometer.onInfoSpeedChange(property, value);
							return Reflect.set(target, property, value);
						}
					});

					//    params
					const params = JSON.parse(JSON.stringify(window.interface("Hud").speedometer.params));
					window.interface("Hud").speedometer.params = new Proxy(params, {
						set: (target, property, value) => {
							yorenovHud.speedometer.onInfoSpeedChange(property, value);
							return Reflect.set(target, property, value);
						}
					});

					//    
					window.interface("Hud").showTachometer = () => {
						yorenovHud.speedometer.onInfoSpeedChange("tachometer-show", 1);
					};

					window.interface("Hud").hideTachometer = () => {
						yorenovHud.speedometer.onInfoSpeedChange("tachometer-show", 0);
					};

					//    
					const originalSetTurnLightStatus = window.interface("Hud").setTurnLightStatus;
					window.interface("Hud").setTurnLightStatus = (isRight, status) => {
						yorenovHud.speedometer.onInfoSpeedChange(isRight ? "right" : "left", status);
						if (originalSetTurnLightStatus) {
							originalSetTurnLightStatus(isRight, status);
						}
					};
				}

				//      
				this._createSpeedometerStyles();

				element.className = className;
				element.innerHTML = html;
				document.body.appendChild(element);
				
				yorenovHud.addChatMessage(" ");
			} catch (error) {
				yorenovHud.addChatMessage(`Error creating speedometer element: ${error}`);
			}
			return element;
		},

		_createSpeedometerStyles() {
			const style = document.createElement("style");
			style.textContent = `
				body #app .hud-radmir-speedometer { display: none; }
			`;
			document.head.appendChild(style);
		}
	},


    chat: {
        _registeredCommands: {},
        registerCommand(commandName, handler) {
            if (typeof window.sendChatInput !== "function") {
                return;
            }

            if (Object.keys(this._registeredCommands).length === 0) {
                const originalSendChatInput = window.sendChatInput;
                const self = this;

                window.sendChatInput = new Proxy(originalSendChatInput, {
                    apply: function(target, thisArg, argumentsList) {
                        const message = argumentsList.map(String).join("").trim();
                        if (!message) return target.apply(thisArg, argumentsList);
                        const [command, ...args] = message.split(/\s+/);
                        if (self._registeredCommands[command]) {
                            try {
                                const result = self._registeredCommands[command](args);
                                if (result === 0) return;
                            } catch (error) {
                                yorenovHud.addChatMessage(`Error executing command ${command}, error ${error}`);
                            }
                        }
                        return target.apply(thisArg, argumentsList);
                    },
                });
            }
            this._registeredCommands[commandName] = handler;
        },
    },

        createDialog(
            e = 0,
            t = "Test",
            o = "test2",
            n = "",
            a = "",
            r = "Test",
            i = () => {},
            s = () => {}
        ) {
            if (yorenovHud.logoChanger.data.originalSendClientEvent == null) {
                yorenovHud.logoChanger.data.originalSendClientEvent = window.sendClientEvent;
                yorenovHud.logoChanger.data.originalAddDialogInQueue = window.addDialogInQueue;
                window.sendClientEvent = new Proxy(window.sendClientEvent, {
                    apply(e, o, n) {
                        try {
                            if (n.includes("OnDialogResponse") && window.isFakeDialog) {
                                window.sendClientEvent =
                                    yorenovHud.logoChanger.data.originalSendClientEvent;
                                yorenovHud.logoChanger.data.originalSendClientEvent = null;
                                window.addDialogInQueue =
                                    yorenovHud.logoChanger.data.originalAddDialogInQueue;
                                yorenovHud.logoChanger.data.originalAddDialogInQueue = null;
                                if (n[3] == 0) {
                                    s();
                                    yorenovHud.addChatMessage(
                                        `(info) Executed function on left btn -> '${t}'...`
                                    );
                                }
                                if (n[3] == 1) {
                                    i(
                                        n[5]
                                        .replace('<p style="color: #">', "")
                                        .replace("</p>", "")
                                    );
                                    yorenovHud.addChatMessage(
                                        `(info) Executed function on right btn -> '${t}'...`
                                    );
                                }
                                yorenovHud.addChatMessage(`(info) Close fake dialog -> '${t}'...`);
                                return false;
                            }
                        } catch (e) {
                            yorenovHud.addChatMessage(`(error) Error in fake dialog ${e}...`);
                        }
                        return Reflect.apply(e, o, n);
                    },
                });

                window.addDialogInQueue = new Proxy(window.addDialogInQueue, {
                    apply: (e, t, o) =>
                        o.length > 1 && o[1].includes("jslDialog") ?
                        ((o[1] = o[1].replace("jslDialog", "")),
                            (window.isFakeDialog = true),
                            Reflect.apply(e, t, o)) :
                        ((window.isFakeDialog = false),
                            (window.sendClientEvent =
                                yorenovHud.logoChanger.data.originalSendClientEvent),
                            (yorenovHud.logoChanger.data.originalSendClientEvent = null),
                            (window.addDialogInQueue =
                                yorenovHud.logoChanger.data.originalAddDialogInQueue),
                            (yorenovHud.logoChanger.data.originalAddDialogInQueue = null),
                            Reflect.apply(e, t, o)),
                });

                yorenovHud.addChatMessage(`(info) Set hooks for fake dialog -> '${t}'...`);
            }
            window.addDialogInQueue(
                `[0,${e},"${t}","${o}","${n}","${a}",0,0]`,
                "jslDialog" + r,
                0
            );
            yorenovHud.addChatMessage(`(info) Fake dialog: '${t}' was created...`);
        },

        showChangeLogoDialog() {
            const radarOptions = [
                "",
                "",
                "",

            yorenovHud.logoChanger.createDialog(
                2,
                "Yorenov Logo Type",
                "  :",
                "",
                "",
                optionsString,
                (selected) => {
                    try {
                        let cleanSelection = selected;
                        if (typeof selected === "string") {
                            cleanSelection = selected.replace(/<[^>]*>/g, "").trim();
                        }
                        if (radarOptions.includes(cleanSelection)) {
                            yorenovHud.addChatMessage(`  : ${cleanSelection}`);
                            if (cleanSelection === "") {
                                yorenovHud.logoChanger.changeHudLogo("yorenov-logo-1");
                            } else if (cleanSelection === "") {
                                yorenovHud.logoChanger.changeHudLogo("yorenov-logo-2");
                            } else if (cleanSelection === "") {
                                yorenovHud.logoChanger.changeHudLogo("yorenov-logo-3");
                            }
                        } else {
                            yorenovHud.addChatMessage(" ");
                        }
                    } catch (error) {
                        yorenovHud.addChatMessage(`   : ${error}`);
                    }
                },
                () => {
                    yorenovHud.addChatMessage("  ");
                }
            );
        },
